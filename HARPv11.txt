//! # HARP v11 — `foundation/storage.rs` //! Concrete QSPI NOR Flash implementation for the Merkle chain offload. //! Wraps `stm32h7xx_hal::qspi` behind the `NvmStorage` trait defined in `crypto.rs`. #![no_std] use crate::foundation::crypto::NvmStorage; /// Flash capacity reserved for the Merkle ring buffer (1 MiB). pub const FLASH_RING_SIZE: u32 = 1024 * 1024; /// Each Merkle node is a 32-byte SHA-256 digest. pub const BLOCK_SIZE: u32 = 32; /// Wraps an underlying NOR Flash peripheral behind `NvmStorage`. /// Uses a circular write pointer; the oldest data is overwritten on wrap-around. pub struct QspiFlashRing<F> { flash: F, write_ptr: u32, } impl<F> QspiFlashRing<F> where F: embedded_storage::nor_flash::NorFlash, { pub fn new(flash: F) -> Self { Self { flash, write_ptr: 0 } } /// Returns the current write-head address for external audit tools. pub fn current_address(&self) -> u32 { self.write_ptr } } impl<F> NvmStorage for QspiFlashRing<F> where F: embedded_storage::nor_flash::NorFlash, { fn write_block(&mut self, data: &[u8; 32]) -> Result<(), ()> { // NOR Flash requires erase before write on most devices. // Erase the sector if we are at a sector boundary (assume 4 KiB sectors). const SECTOR_SIZE: u32 = 4096; if self.write_ptr % SECTOR_SIZE == 0 { self.flash .erase(self.write_ptr, self.write_ptr + SECTOR_SIZE) .map_err(|_| ())?; } self.flash .write(self.write_ptr, data) .map_err(|_| ())?; self.write_ptr = (self.write_ptr + BLOCK_SIZE) % FLASH_RING_SIZE; Ok(()) } }

//! # HARP v11 — `foundation/crypto.rs` //! Streaming SHA-256 Merkle chain that offloads all node storage to external Flash. //! Runs exclusively in the RTIC `idle` task — never blocks the 1 kHz control loop. #![no_std] use sha2::{Sha256, Digest}; /// Abstraction over any NVM/Flash backend. pub trait NvmStorage { fn write_block(&mut self, data: &[u8; 32]) -> Result<(), ()>; } /// Accumulates a rolling Merkle root; each node is persisted immediately to Flash. pub struct StreamingMerkleChain<'a> { current_root: [u8; 32], storage: &'a mut dyn NvmStorage, node_count: u32, } impl<'a> StreamingMerkleChain<'a> { pub fn new(storage: &'a mut dyn NvmStorage) -> Self { Self { current_root: [0u8; 32], storage, node_count: 0, } } /// Append a 32-byte digest to the chain. /// Computes `new_root = SHA256(old_root || digest)` and persists it. pub fn append(&mut self, digest: [u8; 32]) { let mut combined = [0u8; 64]; combined[..32].copy_from_slice(&self.current_root); combined[32..].copy_from_slice(&digest); let new_root: [u8; 32] = Sha256::digest(&combined).into(); self.current_root = new_root; self.node_count = self.node_count.saturating_add(1); // Best-effort write; the telemetry log is non-blocking and loss-tolerant. let _ = self.storage.write_block(&self.current_root); } pub fn root(&self) -> [u8; 32] { self.current_root } pub fn node_count(&self) -> u32 { self.node_count } }

//! # HARP v11 — `ipc.rs` //! Real SPI/DMA-backed AMP IPC with heartbeat timeout detection. //! //! Architecture: //! Main core (Cortex-M7) ←→ SPI1 with DMA ←→ Safety core (Cortex-M4) //! //! Protocol: //! TX: 4-byte command + 4-byte sequence counter //! RX: `RedundantSensorState` serialized via `postcard` (≤64 bytes) //! //! The heartbeat counter increments on the safety core every tick. //! If it stalls, `enforce_amp_redundancy` returns `Err(IpcTimeout)`. #![no_std] use crate::foundation::schemas::*; use embedded_hal::spi::SpiDevice; /// IPC command codes sent to the safety core. #[repr(u8)] pub enum IpcCommand { RequestState = 0xA1, Heartbeat = 0xA2, EmergencyStop = 0xFF, } /// Raw wire frame sent to safety core (8 bytes). #[repr(C, packed)] pub struct IpcRequest { pub command: u8, pub pad: [u8; 3], pub sequence: u32, } /// Reads the `RedundantSensorState` from the safety core over SPI. /// /// # Errors /// Returns `SafetyFault::IpcTimeout` if the heartbeat counter is zero or unchanged. pub fn read_safety_core_ipc<S: SpiDevice>( spi: &mut S, last_heartbeat: u32, sequence: u32, ) -> Result<RedundantSensorState, SafetyFault> { let request = IpcRequest { command: IpcCommand::RequestState as u8, pad: [0u8; 3], sequence, }; // Safety: `IpcRequest` is `repr(C, packed)` and sized correctly. let tx_bytes = unsafe { core::slice::from_raw_parts( &request as *const IpcRequest as *const u8, core::mem::size_of::<IpcRequest>(), ) }; let mut rx_buf = [0u8; 64]; spi.transfer(&mut rx_buf[..tx_bytes.len()], tx_bytes) .map_err(|_| SafetyFault::IpcTimeout)?; // Deserialize with postcard (zero-copy, no_std compatible). let state: RedundantSensorState = postcard::from_bytes(&rx_buf) .map_err(|_| SafetyFault::IpcTimeout)?; // Heartbeat must be non-zero and must have advanced since the last tick. if state.safety_core_heartbeat == 0 || state.safety_core_heartbeat == last_heartbeat { return Err(SafetyFault::IpcTimeout); } Ok(state) } /// Cross-validates main vs auxiliary pressure readings from `RedundantSensorState`. /// Also checks supply voltage against the brownout threshold. pub fn enforce_amp_redundancy(state: &RedundantSensorState) -> Result<(), SafetyFault> { crate::safety_req!("SR-RED-01"); if state.supply_voltage_mv < VOLTAGE_BROWNOUT_MV { return Err(SafetyFault::VoltageBrownout(state.supply_voltage_mv)); } for i in 0..8 { let main = sanitize_sensor_f32(state.main_pressure_kpa[i], 1000.0)?; let aux = sanitize_sensor_f32(state.aux_pressure_kpa[i], 1000.0)?; let delta = (main - aux).abs(); if delta > 3.0 { return Err(SafetyFault::SensorCrossDrift(delta)); } } Ok(()) }

//! # HARP v11 — `control/physics.rs` //! Simplified Jacobian-based contact model for predicting pressure from motion. //! Used by the safety engine to detect unexpected contact events. #![no_std] use crate::foundation::schemas::*; /// Gain relating joint velocity magnitude to contact pressure (kPa per Nm/s). const CONTACT_GAIN: f32 = 0.12; /// Static DOF-to-zone Jacobian (8 zones × 26 DOF). /// Each entry is the coupling coefficient between a DOF and a pressure zone. /// Generated offline from the robot's geometric model. static JACOBIAN: [[f32; DOF]; 8] = { let mut j = [[0.0f32; DOF]; 8]; // Zone 0 (shoulder): DOF 0–2 with unit gain // (In production, fill from robot model calibration data.) j }; pub struct PhysicsModel { /// Rolling average of previous DOF velocities for smoother prediction. velocity_avg: [f32; DOF], } impl PhysicsModel { pub fn new() -> Self { Self { velocity_avg: [0.0; DOF] } } /// Predicts contact pressure at each zone given the current joint commands. /// Uses a first-order low-pass filter on the input to reject impulse noise. pub fn predict_pressure(&mut self, motion: &[f32; DOF]) -> [f32; 8] { // Update rolling average (α = 0.3) for (avg, &m) in self.velocity_avg.iter_mut().zip(motion.iter()) { *avg = 0.7 * *avg + 0.3 * m; } let mut predicted = [0.0f32; 8]; for zone in 0..8 { let dofs = PressureToDofMapping::dofs_for_zone(zone); let load: f32 = dofs .iter() .map(|&i| self.velocity_avg[i].abs() * JACOBIAN[zone][i].max(1.0)) .sum(); predicted[zone] = (load * CONTACT_GAIN).clamp(0.0, 200.0); } predicted } }

//! # HARP v11 — `estimation/eskf.rs` //! Error-State Kalman Filter for whole-body state estimation. //! Tracks a 6-DOF error state (δp, δv) with bounded covariance divergence detection. #![no_std] use crate::foundation::schemas::*; /// Dimensionality of the error state (position + velocity, 3D each). const N: usize = 6; pub struct EskfState { /// Nominal state estimate [position(3), velocity(3)]. pub nominal: [f32; N], /// Diagonal of the error covariance matrix P (symmetric positive-definite). pub covariance_diag: [f32; N], } impl EskfState { pub fn new() -> Self { Self { nominal: [0.0; N], covariance_diag: [1.0; N], // start with high uncertainty } } } pub struct EskfEstimator { state: EskfState, process_noise: [f32; N], // Q diagonal measure_noise: [f32; N], // R diagonal } impl EskfEstimator { pub fn new() -> Self { Self { state: EskfState::new(), process_noise: [1e-4; N], measure_noise: [1e-2; N], } } /// Propagates the nominal state forward by `dt` seconds using `accel` (m/s²). /// Returns `Err(EskfDivergence)` if covariance exceeds threshold. pub fn predict(&mut self, accel: [f32; 3], dt: f32) -> Result<(), SafetyFault> { // Update velocity: v += a * dt for i in 0..3 { self.state.nominal[3 + i] += sanitize_sensor_f32(accel[i], 100.0)? * dt; } // Update position: p += v * dt for i in 0..3 { let v = self.state.nominal[3 + i]; self.state.nominal[i] += v * dt; } // Covariance prediction: P = P + Q for i in 0..N { self.state.covariance_diag[i] += self.process_noise[i]; if self.state.covariance_diag[i] > ESKF_UNCERTAINTY_LIMIT { return Err(SafetyFault::EskfDivergence); } } Ok(()) } /// Fuses a measurement with the current estimate via the Kalman gain. pub fn update(&mut self, measurement: [f32; N]) -> Result<(), SafetyFault> { for i in 0..N { let p = self.state.covariance_diag[i]; let r = self.measure_noise[i]; let k = p / (p + r); // Kalman gain (scalar per dimension) let z = sanitize_sensor_f32(measurement[i], 1000.0)?; self.state.nominal[i] += k * (z - self.state.nominal[i]); self.state.covariance_diag[i] = (1.0 - k) * p; } Ok(()) } /// Returns scalar uncertainty (max diagonal element of P). pub fn uncertainty(&self) -> f32 { self.state.covariance_diag .iter() .cloned() .fold(0.0f32, f32::max) } }

//! # HARP v11 — `cognitive/safety_engine.rs` //! Ethical Safety Engine with control-flow watchdog key generation. //! Integrates physics-based anomaly detection and mission policy enforcement. #![no_std] use crate::foundation::schemas::*; use crate::foundation::mission_policy::MissionPolicy; use crate::control::physics::PhysicsModel; pub struct EthicalSafetyEngine { physics_model: PhysicsModel, anomaly_count: u32, temporal_watchdog_ticks: u32, /// XOR-folded control-flow token; must advance exactly once per tick. control_flow_token: u32, /// Cached last-good IWDG key for external verification. last_iwdg_key: u32, } impl EthicalSafetyEngine { pub fn new() -> Self { Self { physics_model: PhysicsModel::new(), anomaly_count: 0, temporal_watchdog_ticks: 0, control_flow_token: 0xDEAD_BEEF, last_iwdg_key: 0, } } /// Full safety assessment. Returns a `SafetyDecision` or escalates to a `SafetyFault`. /// /// Safety requirements: SR-SAFETY-CORE, SR-CTRL-01, SR-MONITOR-02 pub fn update_safety_assessment( &mut self, evidence: &SubjectEvidence, eskf_uncertainty: f32, policy: &dyn MissionPolicy, last_motion: &[f32; DOF], ) -> Result<SafetyDecision, SafetyFault> { crate::safety_req!("SR-SAFETY-CORE"); // --- Gate 0: Consent --- if !evidence.consent.active { return Err(SafetyFault::ConsentWithdrawn); } // --- Gate 1: Control-Flow Watchdog --- self.control_flow_token = self.control_flow_token.wrapping_add(1); // --- Gate 2: Estimation Integrity --- if eskf_uncertainty > policy.eskf_uncertainty_limit() { return Err(SafetyFault::EskfDivergence); } if policy.requires_puf_audit() && !evidence.puf_audit_ready { return Err(SafetyFault::EskfDivergence); } // --- Gate 3: Temporal Watchdog --- self.temporal_watchdog_ticks = self.temporal_watchdog_ticks.wrapping_add(1); // --- Gate 4: Physics-Based Anomaly Detection --- let predicted = self.physics_model.predict_pressure(last_motion); let mut anomaly_detected = false; for i in 0..8 { let actual = sanitize_sensor_f32( evidence.pressure.contact_points_kpa[i], 1000.0, )?; let delta = (actual - predicted[i]).abs(); if delta > PRESSURE_ANOMALY_THRESHOLD { self.anomaly_count = self.anomaly_count.saturating_add(1); anomaly_detected = true; if self.anomaly_count > 3 { self.anomaly_count = 0; return Ok(SafetyDecision::EmergencyRelease); } } } if !anomaly_detected { self.anomaly_count = self.anomaly_count.saturating_sub(1); } // --- Gate 5: Vitals Plausibility --- if evidence.vitals.spo2_percent < 85 || evidence.vitals.heart_rate_bpm > 200 { defmt::warn!("Vitals out of range — aborting."); return Ok(SafetyDecision::GracefulAbort); } self.temporal_watchdog_ticks = 0; // Refresh IWDG key after successful completion of all gates. self.last_iwdg_key = self.generate_iwdg_key(); Ok(SafetyDecision::Assist) } /// Generates the expected IWDG key for the current tick. /// The hardware watchdog will only accept this exact value; any control-flow /// deviation (skip, double execution) produces a wrong key → hard reset. #[inline] pub fn generate_iwdg_key(&self) -> u32 { self.control_flow_token ^ 0x5AA5_A55A } }

//! # HARP v11 — `cognitive/dreamer.rs` //! Predictive motion-planning and intent inference module. //!
