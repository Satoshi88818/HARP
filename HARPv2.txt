#![no_std] use crate::foundation::schemas::*; // [StabilityMonitor and PowerClamp remain as in v1] // ============================================================================= // Improvement 1: Impedance / Force-Compliance Control Layer // // Transitions the WBC from pure position control to force-aware "give-way" // behaviour. If a torque sensor at a contact point (hand, chest, forearm) // exceeds the biological safety threshold, the impedance controller overrides // the position target and yields to the contact force. // ============================================================================= /// Tracks the virtual spring-damper state for each of the primary contact DOFs. pub struct ImpedanceController { /// Estimated contact force per DOF (N), updated from torque sensors each tick pub contact_force_n: [f32; DOF], /// Virtual displacement from the impedance spring (m) pub virtual_displacement_m: [f32; DOF], /// Virtual velocity for the damper term (m/s) pub virtual_velocity_ms: [f32; DOF], } impl ImpedanceController { pub fn new() -> Self { Self { contact_force_n: [0.0; DOF], virtual_displacement_m: [0.0; DOF], virtual_velocity_ms: [0.0; DOF], } } /// Update contact force estimates from raw torque sensor readings. /// `torque_sensor_nm` is the measured joint torque; `link_length_m` converts /// torque to approximate end-effector force. pub fn update_contact_forces( &mut self, torque_sensor_nm: &[f32; DOF], link_length_m: &[f32; DOF], ) { for i in 0..DOF { if link_length_m[i] > 1e-4 { self.contact_force_n[i] = torque_sensor_nm[i] / link_length_m[i]; } } } /// Apply impedance compliance: modify commanded velocities so that the robot /// "gives way" if contact force exceeds the biological safety threshold. /// /// Returns a corrected command vector. The correction uses a virtual /// spring-damper model: /// F_virtual = K * x_displacement + D * x_dot /// If F_contact > limit, override command with a retraction velocity. pub fn apply_impedance_compliance( &mut self, cmd: &mut [f32; DOF], contact_limit_n: f32, dt_s: f32, ) { for i in 0..DOF { let f = self.contact_force_n[i].abs(); if f > contact_limit_n { // Force exceeds biological safety limit — compute retraction let excess = f - contact_limit_n; // Spring-damper retraction: retract proportional to excess force let retract_velocity = -(excess / IMPEDANCE_STIFFNESS) - IMPEDANCE_DAMPING * self.virtual_velocity_ms[i]; // Clamp retraction to prevent violent jerks let retract_clamped = retract_velocity.max(-0.5).min(0.0); // Override the position-control command with compliance command cmd[i] = retract_clamped; // Update virtual state self.virtual_displacement_m[i] += retract_clamped * dt_s; self.virtual_velocity_ms[i] = (self.virtual_velocity_ms[i] * 0.8) + (retract_clamped * 0.2); defmt::warn!( "WBC Impedance: Joint {} exceeded {}N limit ({}N) — complying", i, contact_limit_n, f ); } else { // Decay virtual state back to zero when contact is released self.virtual_displacement_m[i] *= 0.95; self.virtual_velocity_ms[i] *= 0.9; } } } } // ============================================================================= // ThermalSolver — updated with impedance compliance + graceful degradation // ============================================================================= pub struct ThermalSolver; impl ThermalSolver { /// Solve joint commands with: /// 1. Impedance compliance (force give-way) — Improvement 1 /// 2. Sacrificial thermal overload for life-critical maneuvers /// 3. Failed-joint zeroing for Graceful Degradation — Improvement 3 pub fn solve_with_rescue_priority( target_v: &[f32; DOF], physics: &PhysicalConstraints, life_in_danger: bool, impedance: &mut ImpedanceController, health: &ComponentHealth, dt_s: f32, ) -> [f32; DOF] { let mut cmd = *target_v; // Step 1: Apply impedance compliance before thermal gating // Use chest/chassis limit as the conservative bound for whole-body contact impedance.apply_impedance_compliance( &mut cmd, CONTACT_FORCE_CHEST_LIMIT_N, dt_s, ); // Step 2: Thermal de-rating (Sacrificial Overload when LifeInDanger) let limit = if life_in_danger { JOINT_THERMAL_SACRIFICIAL_C } else { JOINT_THERMAL_WARNING_C }; for i in 0..DOF { let temp = physics.joint_temperatures_c[i]; if temp > limit { let overheat = (temp - limit).max(0.0); cmd[i] /= (1.0 + overheat).sqrt(); } } // Step 3: Improvement 3 — Zero out commands for failed/sacrificed joints. // The Dreamer's kinematic re-mapping already accounts for these DOFs; // this ensures we never command a non-functional actuator. for i in 0..DOF { if health.is_failed(i) { cmd[i] = 0.0; } } cmd } } /// Placeholder for physical constraint data read from the HAL pub struct PhysicalConstraints { pub joint_temperatures_c: [f32; DOF], pub torque_sensor_nm: [f32; DOF], pub link_length_m: [f32; DOF], }
